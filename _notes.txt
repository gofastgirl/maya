How to Make API calls! Can copy and paste this...
-------------------------------------------
using (APIcall api = new APIcall())
        {
            // --- OPNsense Platform Example ---
            IAuthenticator platformAuth = new platformAuthenticator(platformApiKey, platformApiSecret);
            if (api.ConnectToPlatform(Ip, UseHttps, platformAuth))
            {
                try
                {
                    string systemStatusJson = await api.GetData("/api/core/system/status");
                    string? SystemName = api.GetValue(systemStatusJson, "SystemName");
                    string? version = api.GetValue(systemStatusJson, "system.firmware.version");
                }
                catch (Exception ex)
                {
                    throw; // Re-throw to propagate the error
                }
            }
            else
            {
                throw new InvalidOperationException("Failed to connect to OPNsense.");
            }
    }
